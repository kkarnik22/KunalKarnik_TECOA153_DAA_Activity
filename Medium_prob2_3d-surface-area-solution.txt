# Complete the surfaceArea function below.
def surfaceArea(A):
    #A=[[1 3 4][2 2 3][1 2 4]]
    area=0

    for i in range(len(A)):
        
        for j in range(len(A[i])):
            
            for z in range(A[i][j]):
                

                if i-1<0:
                    area+=1
                else: 
                    if A[i-1][j]<=z:
                        area+=1
                if i+1>=len(A):
                    area+=1
                else: 
                    if A[i+1][j]<=z:
                        area+=1
               
                if j-1<0:
                    area+=1
                else: 
                    if A[i][j-1]<=z:
                        area+=1 
                if j+1>=len(A[i]):
                    area+=1
                else: 
                    if A[i][j+1]<=z:
                        area+=1
                if z-1<0:
                    area+=1
                if z+1>=A[i][j]:
                    area+=1
        
                
    return area
